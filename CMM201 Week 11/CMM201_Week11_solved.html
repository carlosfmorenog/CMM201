## Plotting a scatterplot with a legend

setosa = plt.scatter(data[0:50][:,0],data[0:50][:,1],c='b',s=data[0:50][:,2]*10)
versicolor = plt.scatter(data[50:100][:,0],data[50:100][:,1],c='r',s=data[50:100][:,2]*10)
